const tray = document.getElementById('tray');

const stepCurrentLabel = document.getElementById('step-current');

const trayLabels = ['Identificação','Consumo','Avaliação','Feedback'];

function renderTray(){

  tray.innerHTML = '';

  for(let i=0;i<totalSteps;i++){

    const wrap = document.createElement('div');

    wrap.className = 'ball-wrap';

    const ball = document.createElement('div');

    ball.className = 'ball' + (i < state.step ? ' done' : '') + (i === state.step ? ' current' : '');

    const label = document.createElement('div');

    label.className = 'tray-label';

    label.textContent = trayLabels[i];

    wrap.appendChild(ball);

    wrap.appendChild(label);

    tray.appendChild(wrap);

    if(i < totalSteps - 1){

      const line = document.createElement('div');

      line.className = 'tray-line';

      tray.appendChild(line);

    }

  }

  stepCurrentLabel.textContent = state.step + 1;

}

function clearFieldError(el){

  el.classList.remove('error');

}

function renderStep(){

  renderTray();

  card.innerHTML = '';

  if(state.step === 0) renderStep1();

  else if(state.step === 1) renderStep2();

  else if(state.step === 2) renderStep3();

  else if(state.step === 3) renderStep4();

}

function makeNav(onNext, nextLabel){

  const nav = document.createElement('div');

  nav.className = 'nav-row';

  const back = document.createElement('button');

  back.className = 'nav-btn btn-back';

  back.textContent = '← Voltar';

  back.disabled = state.step === 0;

  back.onclick = () => { state.step--; renderStep(); };

  const next = document.createElement('button');

  next.className = 'nav-btn btn-next';

  next.textContent = nextLabel || 'Continuar →';

  next.onclick = onNext;

  nav.appendChild(back);

  nav.appendChild(next);

  return nav;

}

}

function renderStep2(){

  const list = document.createElement('div');

  list.className = 'check-list';

  produtoOpcoes.forEach((opt, idx) => {

    const item = document.createElement('label');

    item.className = 'check-item' + (state.produtos.includes(opt) ? ' checked' : '');

    item.innerHTML = `

      <input type="checkbox" ${state.produtos.includes(opt) ? 'checked' : ''}>

      <span class="check-text">${opt}</span>

    `;

    item.onclick = (e) => {

      e.preventDefault();

      const i = state.produtos.indexOf(opt);

      if(i > -1) state.produtos.splice(i,1);

      else state.produtos.push(opt);

      item.classList.toggle('checked');

      item.querySelector('input').checked = state.produtos.includes(opt);

      if(state.produtos.length) errorMsg.style.display = 'none';

    };

    list.appendChild(item);

  });

  card.appendChild(list);

  card.appendChild(errorMsg);

  card.appendChild(makeNav(() => {

    if(state.produtos.length === 0){

      errorMsg.style.display = 'block';

      return;

    }

    state.step++;

    renderStep();

  }));

}

}

function formatDocumento(v){

  const d = v.replace(/\D/g,'').slice(0,14);

  if(d.length <= 11){

    return d

      .replace(/(\d{3})(\d)/,'$1.$2')

      .replace(/(\d{3})(\d)/,'$1.$2')

      .replace(/(\d{3})(\d{1,2})$/,'$1-$2');

  }

  return d

    .replace(/(\d{2})(\d)/,'$1.$2')

    .replace(/(\d{3})(\d)/,'$1.$2')

    .replace(/(\d{3})(\d)/,'$1/$2')

    .replace(/(\d{4})(\d{1,2})$/,'$1-$2');

}

function escapeHtml(str){

  return (str || '').replace(/[&<>"']/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]));

}
